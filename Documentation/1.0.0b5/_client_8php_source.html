706  public function close()
707  {
708  $result = true;
709  /*
710  * The check below is done because for some unknown reason
711  * (either a PHP or a RouterOS bug) calling "/quit" on an encrypted
712  * connection makes one end hang.
713  *
714  * Since encrypted connections only appeared in RouterOS 6.1, and
715  * the "/quit" call is needed for all <6.0 versions, problems due
716  * to its absence should be limited to some earlier 6.* versions
717  * on some RouterBOARD devices.
718  */
719  if ($this->com->getTransmitter()->getCrypto() === N::CRYPTO_OFF) {
720  if (null !== $this->registry) {
721  $this->registry->setTaglessMode(true);
722  }
723  try {
724  $response = $this->sendSync(new Request('/quit'));
725  $result = $response[0]->getType() === Response::TYPE_FATAL;
726  } catch (SocketException $e) {

733  $this->registry->setTaglessMode(false);
734  }
735  }
