182  /**
183  * Sets a property to be usable as a key in the collection.
184  *
185  * @param string|null $name The name of the property to use. Future calls
186  * that accept a position will then also be able to search values of
187  * that property for a matching value.
188  * Specifying NULL will disable such lookups (as is by default).
189  * Note that in case this value occurs multiple times within the
190  * collection, only the last matching response will be accessible by
191  * that value.
192  *
193  * @return $this The object itself.

196  {

198  $name = (string)$name;

200  $this->responsesIndex[$name] = array();
201  foreach ($this->responses as $pos => $response) {
202  $val = $response->getProperty($name);
203  if (null !== $val) {
204  $this->responsesIndex[$name][$val] = $pos;
205  }
206  }
207  }
208  }
209  $this->index = $name;
210  return $this;
211  }

488  /**
489  * Order resones by criteria.
490  *
491  * @param string[]|array<string,null|int|array<int|callable>> $criteria The
492  * criteria to order responses by. It takes the
493  * form of an array where each key is the name of the property to use
494  * as (N+1)th sorting key. The value of each member can be either NULL
495  * (for that property, sort normally in ascending order), a single sort
496  * order constant (SORT_ASC or SORT_DESC) to sort normally in the
497  * specified order, an array where the first member is an order
498  * constant, and the second one is sorting flags (same as built in PHP
499  * array functions) or a callback.
500  * If a callback is provided, it must accept two arguments
501  * (the two values to be compared), and return -1, 0 or 1 if the first
502  * value is respectively less than, equal to or greater than the second
503  * one.
504  * Each key of $criteria can also be numeric, in which case the
505  * value is the name of the property, and sorting is done normally in
506  * ascending order.
507  *
508  * @return static A new collection with the responses sorted in the
509  * specified order.
510  */
511  public function orderBy(array $criteria)
512  {
513  $this->compareBy = $criteria;
514  $sortedResponses = $this->responses;
515  usort($sortedResponses, array($this, 'compare'));
516  return new static($sortedResponses);

539  /**
540  * Compares two responses.
541  *
542  * Compares two responses, based on criteria defined in
543  * {@link static::$compareBy}.
544  *
545  * @param Response $itemA The response to compare.
546  * @param Response $itemB The response to compare $a against.
547  *
548  * @return int Returns 0 if the two responses are equal according to every
549  * criteria specified, -1 if $a should be placed before $b, and 1 if $b
550  * should be placed before $a.
551  */
552  protected function compare(Response $itemA, Response $itemB)
553  {
554  foreach ($this->compareBy as $name => $spec) {
555  if (!is_string($name)) {
556  $name = $spec;
557  $spec = null;
558  }
559 
560  $members = array(
561  0 => $itemA->getProperty($name),
562  1 => $itemB->getProperty($name)
563  );
564 
565  if (is_callable($spec)) {
566  uasort($members, $spec);
567  } elseif ($members[0] === $members[1]) {
568  continue;
569  } else {
570  $flags = SORT_REGULAR;
571  $order = SORT_ASC;
572  if (is_array($spec)) {
573  list($order, $flags) = $spec;
574  } elseif (null !== $spec) {
575  $order = $spec;
576  }
577 
578  if (SORT_ASC === $order) {
579  asort($members, $flags);
580  } else {
581  arsort($members, $flags);
582  }
583  }
584  return (key($members) === 0) ? -1 : 1;
585  }
586 
587  return 0;
588  }
