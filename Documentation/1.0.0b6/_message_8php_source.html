68  /**
69  * A shorthand gateway.
70  *
71  * This is a magic PHP method that allows you to call the object as a
72  * function. Depending on the argument given, one of the other functions in
73  * the class is invoked and its returned value is returned by this function.
74  *
75  * @param string|null $name The name of an attribute to get the value of,
76  * or NULL to get the tag.
77  *
78  * @return string|resource The value of the specified attribute,
79  * or the tag if NULL is provided.
80  */

83  if (null === $name) {
84  return $this->getTag();

89  /**

96  public static function sanitizeAttributeName($name)
97  {
98  $name = (string) $name;
99  if ((empty($name) && $name !== '0')
100  || preg_match('/[=\s]/s', $name)
101  ) {
102  throw new InvalidArgumentException(
103  'Invalid name of argument supplied.',
104  InvalidArgumentException::CODE_NAME_INVALID

106  }

117  public static function sanitizeAttributeValue($value)

199  /**
200  * Sets an attribute for the message.
201  *
202  * @param string $name Name of the attribute.
203  * @param string|resource|null $value Value of the attribute as a string or
204  * seekable stream.
205  * Setting the value to NULL removes an argument of this name.
206  * If a seekable stream is provided, it is sent from its current
207  * position to its end, and the pointer is seeked back to its current
208  * position after sending.
209  * Non seekable streams, as well as all other types, are casted to a
210  * string.
211  *
212  * @return $this The message object.
213  *
214  * @see getArgument()
215  */
216  protected function setAttribute($name, $value = '')
217  {
218  if (null === $value) {
219  unset($this->attributes[self::sanitizeAttributeName($name)]);
220  } else {
221  $this->attributes[self::sanitizeAttributeName($name)]
222  = self::sanitizeAttributeValue($value);
223  }
224  return $this;
225  }
