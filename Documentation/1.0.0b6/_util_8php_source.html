161  /**
162  * Creates a Request object.
163  *
164  * Creates a {@link Request} object, with a command that's at the
165  * current menu. The request can then be sent using {@link Client}.
166  *
167  * @param string $command The command of the request, not including
168  * the menu. The request will have that command at the current menu.
169  * @param array $args Arguments of the request.
170  * Each array key is the name of the argument, and each array value is
171  * the value of the argument to be passed.
172  * Arguments without a value (i.e. empty arguments) can also be
173  * specified using a numeric key, and the name of the argument as the
174  * array value.
175  * @param Query|null $query The {@link Query} of the request.
176  * @param string|null $tag The tag of the request.
177  *
178  * @return Request The {@link Request} object.
179  *
180  * @throws NotSupportedException On an attempt to call a command in a
181  * different menu using API syntax.
182  * @throws InvalidArgumentException On an attempt to call a command in a
183  * different menu using CLI syntax.
184  */
185  public function newRequest(
186  $command,
187  array $args = array(),
188  Query $query = null,
189  $tag = null
190  ) {
191  if (false !== strpos($command, '/')) {
192  throw new NotSupportedException(
193  'Command tried to go to a different menu',
194  NotSupportedException::CODE_MENU_MISMATCH,
195  null,
196  $command
197  );
198  }
199  $request = new Request('/menu', $query, $tag);
200  $request->setCommand("{$this->menu}/{$command}");
201  foreach ($args as $name => $value) {
202  if (is_int($name)) {
203  $request->setArgument($value);
204  } else {
205  $request->setArgument($name, $value);
206  }
207  }
208  return $request;
209  }

211  /**
212  * Executes a RouterOS script.
213  *
214  * Executes a RouterOS script, written as a string or a stream.
215  * Note that in cases of errors, the line numbers will be off, because the
216  * script is executed at the current menu as context, with the specified
217  * variables pre declared. This is achieved by prepending 1+count($params)
218  * lines before your actual script.
219  *
220  * @param string|resource $source The source of the script,
221  * as a string or stream. If a stream is provided, reading starts from
222  * the current position to the end of the stream, and the pointer stays
223  * at the end after reading is done.
224  * @param array<string|int,mixed> $params An array of parameters to make
225  * available in the script as local variables.
226  * Variable names are array keys, and variable values are array values.
227  * Array values are automatically processed with
228  * {@link static::escapeValue()}. Streams are also supported, and are
229  * processed in chunks, each with
230  * {@link static::escapeString()} with all bytes being escaped.
231  * Processing starts from the current position to the end of the stream,
232  * and the stream's pointer is left untouched after the reading is done.
233  * Variables with a value of type "nothing" can be declared with a
234  * numeric array key and the variable name as the array value
235  * (that is casted to a string).
236  * Note that the script's (generated) name is always added as the
237  * variable "_", which will be inadvertently lost if you overwrite it
238  * from here.
239  * @param string|null $policy Allows you to specify a policy the
240  * script must follow. Has the same format as in terminal.
241  * If left NULL, the script has no restrictions beyond those imposed by
242  * the username.
243  * @param string|null $name The script is executed after being
244  * saved in "/system script" and is removed after execution.
245  * If this argument is left NULL, a random string,
246  * prefixed with the computer's name, is generated and used
247  * as the script's name.
248  * To eliminate any possibility of name clashes,
249  * you can specify your own name instead.
250  *
251  * @return ResponseCollection The responses of all requests involved, i.e.
252  * the add, the run and the remove.
253  *
254  * @throws RouterErrorException When there is an error in any step of the
255  * way. The reponses include all successful commands prior to the error
256  * as well. If the error occurs during the run, there will also be a
257  * remove attempt, and the results will include its results as well.
258  */
259  public function exec(
260  $source,
261  array $params = array(),
262  $policy = null,
263  $name = null
264  ) {
265  if (null === $name) {
266  $name = uniqid(gethostname(), true);
267  }
268 
269  $request = new Request('/system/script/add');
270  $request->setArgument('name', $name);
271  $request->setArgument('policy', $policy);
272 
273  $params += array('_' => $name);
274 
275  $finalSource = fopen('php://temp', 'r+b');
276  fwrite(
277  $finalSource,
278  '/' . str_replace('/', ' ', substr($this->menu, 1)). "\n"
279  );
280  Script::append($finalSource, $source, $params);
281  fwrite($finalSource, "\n");
282  rewind($finalSource);
283 
284  $request->setArgument('source', $finalSource);
285  $addResult = $this->client->sendSync($request);
286 
287  if (count($addResult->getAllOfType(Response::TYPE_ERROR)) > 0) {
288  throw new RouterErrorException(
289  'Error when trying to add script',
290  RouterErrorException::CODE_SCRIPT_ADD_ERROR,
291  null,
292  $addResult
293  );
294  }
295 
296  $request = new Request('/system/script/run');
297  $request->setArgument('number', $name);
298  $runResult = $this->client->sendSync($request);
299  $request = new Request('/system/script/remove');
300  $request->setArgument('numbers', $name);
301  $removeResult = $this->client->sendSync($request);
302 
303  $results = new ResponseCollection(
304  array_merge(
305  $addResult->toArray(),
306  $runResult->toArray(),
307  $removeResult->toArray()
308  )
309  );
310 
311  if (count($runResult->getAllOfType(Response::TYPE_ERROR)) > 0) {
312  throw new RouterErrorException(
313  'Error when running script',
314  RouterErrorException::CODE_SCRIPT_RUN_ERROR,
315  null,
316  $results
317  );
318  }
319  if (count($removeResult->getAllOfType(Response::TYPE_ERROR)) > 0) {
320  throw new RouterErrorException(
321  'Error when removing script',
322  RouterErrorException::CODE_SCRIPT_REMOVE_ERROR,
323  null,
324  $results
325  );
326  }
327 
328  return $results;
329  }

331  /**
332  * Clears the ID cache.
333  *
334  * Normally, the ID cache improves performance when targeting items by a
335  * number. If you're using both Util's methods and other means (e.g.
336  * {@link Client} or {@link Util::exec()}) to add/move/remove items, the

342  * {@link static::add()} adds the new item's ID to the cache as the next

346  * use {@link static::find()} with a callback.

352  $this->idCache = null;
353  return $this;

356  /**
357  * Gets the current time on the router.
358  *
359  * Gets the current time on the router, regardless of the current menu.
360  *
361  * If the timezone is one known to both RouterOS and PHP, it will be used
362  * as the timezone identifier. Otherwise (e.g. "manual"), the current GMT
363  * offset will be used as a timezone, without any DST awareness.
364  *
365  * @return DateTime The current time of the router, as a DateTime object.
366  */
367  public function getCurrentTime()
368  {
369  $clock = $this->client->sendSync(
370  new Request(
371  '/system/clock/print
372  .proplist=date,time,time-zone-name,gmt-offset'
373  )
374  )->current();
375  $clockParts = array();
376  foreach (array(
377  'date',
378  'time',
379  'time-zone-name',
380  'gmt-offset'
381  ) as $clockPart) {
382  $clockParts[$clockPart] = $clock->getProperty($clockPart);
383  if (Stream::isStream($clockParts[$clockPart])) {
384  $clockParts[$clockPart] = stream_get_contents(
385  $clockParts[$clockPart]
386  );
387  }
388  }
389  $datetime = ucfirst(strtolower($clockParts['date'])) . ' ' .
390  $clockParts['time'];
391  try {
392  $result = DateTime::createFromFormat(
393  'M/j/Y H:i:s',
394  $datetime,
395  new DateTimeZone($clockParts['time-zone-name'])
396  );
397  } catch (E $e) {
398  $result = DateTime::createFromFormat(
399  'M/j/Y H:i:s P',
400  $datetime . ' ' . $clockParts['gmt-offset'],
401  new DateTimeZone('UTC')
402  );
403  }
404  return $result;
405  }

407  /**
408  * Finds the IDs of items at the current menu.
409  *
410  * Finds the IDs of items based on specified criteria, and returns them as
411  * a comma separated string, ready for insertion at a "numbers" argument.
412  *
413  * Accepts zero or more criteria as arguments. If zero arguments are
414  * specified, returns all items' IDs. The value of each criteria can be a
415  * number (just as in Winbox), a literal ID to be included, a {@link Query}
416  * object, or a callback. If a callback is specified, it is called for each
417  * item, with the item as an argument. If it returns a true value, the
418  * item's ID is included in the result. Every other value is casted to a
419  * string. A string is treated as a comma separated values of IDs, numbers
420  * or callback names. Non-existent callback names are instead placed in the
421  * result, which may be useful in menus that accept identifiers other than
422  * IDs, but note that it can cause errors on other menus.
423  *
424  * @return string A comma separated list of all items matching the
425  * specified criteria.
426  */
427  public function find()
428  {
429  if (func_num_args() === 0) {
430  if (null === $this->idCache) {
431  $ret = $this->client->sendSync(
432  new Request($this->menu . '/find')
433  )->getProperty('ret');
434  if (null === $ret) {
435  $this->idCache = array();
436  return '';
437  } elseif (!is_string($ret)) {
438  $ret = stream_get_contents($ret);
439  }
440 
441  $idCache = str_replace(
442  ';',
443  ',',
444  strtolower($ret)
445  );
446  $this->idCache = explode(',', $idCache);
447  return $idCache;
448  }
449  return implode(',', $this->idCache);
450  }
451  $idList = '';
452  foreach (func_get_args() as $criteria) {
453  if ($criteria instanceof Query) {
454  foreach ($this->client->sendSync(
455  new Request($this->menu . '/print .proplist=.id', $criteria)
456  )->getAllOfType(Response::TYPE_DATA) as $response) {
457  $newId = $response->getProperty('.id');
458  $idList .= strtolower(
459  is_string($newId)
460  ? $newId
461  : stream_get_contents($newId) . ','
462  );
463  }
464  } elseif (is_callable($criteria)) {
465  $idCache = array();
466  foreach ($this->client->sendSync(
467  new Request($this->menu . '/print')
468  )->getAllOfType(Response::TYPE_DATA) as $response) {
469  $newId = $response->getProperty('.id');
470  $newId = strtolower(
471  is_string($newId)
472  ? $newId
473  : stream_get_contents($newId)
474  );
475  if ($criteria($response)) {
476  $idList .= $newId . ',';
477  }
478  $idCache[] = $newId;
479  }
480  $this->idCache = $idCache;
481  } else {
482  $this->find();
483  if (is_int($criteria)) {
484  if (isset($this->idCache[$criteria])) {
485  $idList = $this->idCache[$criteria] . ',';
486  }
487  } else {
488  $criteria = (string)$criteria;
489  if ($criteria === (string)(int)$criteria) {
490  if (isset($this->idCache[(int)$criteria])) {
491  $idList .= $this->idCache[(int)$criteria] . ',';
492  }
493  } elseif (false === strpos($criteria, ',')) {
494  $idList .= $criteria . ',';
495  } else {
496  $criteriaArr = explode(',', $criteria);
497  for ($i = count($criteriaArr) - 1; $i >= 0; --$i) {
498  if ('' === $criteriaArr[$i]) {
499  unset($criteriaArr[$i]);
500  } elseif ('*' === $criteriaArr[$i][0]) {
501  $idList .= $criteriaArr[$i] . ',';
502  unset($criteriaArr[$i]);
503  }
504  }
505  if (!empty($criteriaArr)) {
506  $idList .= call_user_func_array(
507  array($this, 'find'),
508  $criteriaArr
509  ) . ',';
510  }
511  }
512  }
513  }
514  }
515  return rtrim($idList, ',');
516  }

518  /**
519  * Gets a value of a specified item at the current menu.
520  *
521  * @param int|string|null|Query $number A number identifying the target
522  * item. Can also be an ID or (in some menus) name. For menus where
523  * there are no items (e.g. "/system identity"), you can specify NULL.
524  * You can also specify a {@link Query}, in which case the first match
525  * will be considered the target item.
526  * @param string|resource|null $valueName The name of the value to get.
527  * If omitted, or set to NULL, gets all properties of the target item.
528  *
529  * @return string|resource|null|array The value of the specified
530  * property as a string or as new PHP temp stream if the underlying
531  * {@link Client::isStreamingResponses()} is set to TRUE.
532  * If the property is not set, NULL will be returned.
533  * If $valueName is NULL, returns all properties as an array, where
534  * the result is parsed with {@link Script::parseValueToArray()}.
535  *
536  * @throws RouterErrorException When the router returns an error response
537  * (e.g. no such item, invalid property, etc.).
538  */
539  public function get($number, $valueName = null)
540  {
541  if ($number instanceof Query) {
542  $number = explode(',', $this->find($number));
543  $number = $number[0];
544  } elseif (is_int($number) || ((string)$number === (string)(int)$number)) {
545  $this->find();
546  if (isset($this->idCache[(int)$number])) {
547  $number = $this->idCache[(int)$number];
548  } else {
549  throw new RouterErrorException(
550  'Unable to resolve number from ID cache (no such item maybe)',
551  RouterErrorException::CODE_CACHE_ERROR
552  );
553  }
554  }
555 
556  $request = new Request($this->menu . '/get');
557  $request->setArgument('number', $number);
558  $request->setArgument('value-name', $valueName);
559  $responses = $this->client->sendSync($request);
560  if (Response::TYPE_ERROR === $responses->getType()) {
561  throw new RouterErrorException(
562  'Error getting property',
563  RouterErrorException::CODE_GET_ERROR,
564  null,
565  $responses
566  );
567  }
568 
569  $result = $responses->getProperty('ret');
570  if (Stream::isStream($result)) {
571  $result = stream_get_contents($result);
572  }
573  if (null === $valueName) {
574  // @codeCoverageIgnoreStart
575  //Some earlier RouterOS versions use "," instead of ";" as separator
576  //Newer versions can't possibly enter this condition
577  if (false === strpos($result, ';')
578  && preg_match('/^([^=,]+\=[^=,]*)(?:\,(?1))+$/', $result)
579  ) {
580  $result = str_replace(',', ';', $result);
581  }
582  // @codeCoverageIgnoreEnd
583  return Script::parseValueToArray('{' . $result . '}');
584  }
585  return $result;
586  }

673  /**
674  * Comments items.
675  *
676  * Sets new comments on all items at the current menu
677  * which match certain criteria, using the "comment" command.
678  *
679  * Note that not all menus have a "comment" command. Most notably, those are
680  * menus without items in them (e.g. "/system identity"), and menus with
681  * fixed items (e.g. "/ip service").
682  *
683  * @param mixed $numbers Targeted items. Can be any criteria
684  * accepted by {@link static::find()}.
685  * @param string|resource $comment The new comment to set on the item as a
686  * string or a seekable stream.
687  * If a seekable stream is provided, it is sent from its current
688  * position to its end, and the pointer is seeked back to its current
689  * position after sending.
690  * Non seekable streams, as well as all other types, are casted to a
691  * string.
692  *
693  * @return ResponseCollection Returns the response collection, allowing you
694  * to inspect the output. Current RouterOS versions don't return

1036  /**
1037  * Puts a file on RouterOS's file system.
1038  *
1039  * Puts a file on RouterOS's file system, regardless of the current menu.
1040  * Note that this is a **VERY VERY VERY** time consuming method - it takes a
1041  * minimum of a little over 4 seconds, most of which are in sleep. It waits

1043  * writing to the file), and another 2 seconds after its contents is written

1046  * sleep, used to verify the file was really deleted.

1050  * {@link static::setMenu()} to move to the "/file" menu, and call
1051  * {@link static::remove()} without performing verification afterwards.
1052  *

1054  * @param string|resource|null $data The data the file is going to have
1055  * as a string or a seekable stream.

1057  * If a seekable stream is provided, it is sent from its current
1058  * position to its end, and the pointer is seeked back to its current
1059  * position after sending.
1060  * Non seekable streams, as well as all other types, are casted to a
1061  * string.

1063  * it exists.

1066  */

1072  );

1074

1076  if (!$fileExists) {
1077  return false;

1089  return false;

1092  $printRequest->setArgument('file', $filename)
