83  *
84  * @param Communicator $com The communicator from which to extract
85  * the new response.
86  * @param bool $asStream Whether to populate the argument values
87  * with streams instead of strings.
88  * @param int $sTimeout If a response is not immediately
89  * available, wait this many seconds. If NULL, wait indefinitely.
90  * @param int|null $usTimeout Microseconds to add to the waiting time.
91  * @param Registry|null $reg An optional registry to sync the
92  * response with.
93  *
94  * @see getType()
95  * @see getArgument()
96  */
97  public function __construct(
98  Communicator $com,
99  $asStream = false,
100  $sTimeout = 0,
101  $usTimeout = null,
102  Registry $reg = null
103  ) {
104  if (null === $reg) {
105  if ($com->getTransmitter()->isPersistent()) {
106  $old = $com->getTransmitter()
107  ->lock(T\Stream::DIRECTION_RECEIVE);
108  try {
109  $this->_receive($com, $asStream, $sTimeout, $usTimeout);
110  } catch (E $e) {
111  $com->getTransmitter()->lock($old, true);
112  throw $e;
113  }
114  $com->getTransmitter()->lock($old, true);
115  } else {
116  $this->_receive($com, $asStream, $sTimeout, $usTimeout);
117  }
118  } else {
119  while (null === ($response = $reg->getNextResponse())) {
120  $newResponse = new self($com, true, $sTimeout, $usTimeout);
121  $tagInfo = $reg::parseTag($newResponse->getTag());
122  $newResponse->setTag($tagInfo[1]);
123  if (!$reg->add($newResponse, $tagInfo[0])) {
124  $response = $newResponse;
125  break;
126  }
127  }
128 
129  $this->_type = $response->_type;
130  $this->attributes = $response->attributes;
131  $this->unrecognizedWords = $response->unrecognizedWords;
132  $this->setTag($response->getTag());
133 
134  if (!$asStream) {
135  foreach ($this->attributes as $name => $value) {
136  $this->setAttribute(
137  $name,
138  stream_get_contents($value)
139  );
140  }
141  foreach ($response->unrecognizedWords as $i => $value) {
142  $this->unrecognizedWords[$i] = stream_get_contents($value);
143  }
144  }
145  }

148  /**
149  * Extracts a new response from a communicator.
150  *
151  * This is the function that performs the actual receiving, while the
152  * constructor is also involved in locks and registry sync.
153  *
154  * @param Communicator $com The communicator from which to extract
155  * the new response.
156  * @param bool $asStream Whether to populate the argument values
157  * with streams instead of strings.
158  * @param int $sTimeout If a response is not immediately
159  * available, wait this many seconds. If NULL, wait indefinitely.
160  * Note that if an empty sentence is received, the timeout will be
161  * reset for another sentence receiving.
162  * @param int|null $usTimeout Microseconds to add to the waiting time.
163  *
164  * @return void
165  */
166  private function _receive(
167  Communicator $com,
168  $asStream = false,
169  $sTimeout = 0,
170  $usTimeout = null
171  ) {
172  do {
173  if (!$com->getTransmitter()->isDataAwaiting(
174  $sTimeout,
175  $usTimeout
176  )) {
177  throw new SocketException(
178  'No data within the time limit',
179  SocketException::CODE_NO_DATA
180  );
181  }
182  $type = $com->getNextWord();
183  } while ('' === $type);
184  $this->setType($type);
185  if ($asStream) {
186  for ($word = $com->getNextWordAsStream(), fseek($word, 0, SEEK_END);
187  ftell($word) !== 0;
188  $word = $com->getNextWordAsStream(), fseek(
189  $word,
190  0,
191  SEEK_END
192  )) {
193  rewind($word);
194  $ind = fread($word, 1);
195  if ('=' === $ind || '.' === $ind) {
196  $prefix = stream_get_line($word, null, '=');
197  }
198  if ('=' === $ind) {
199  $value = fopen('php://temp', 'r+b');
200  $bytesCopied = ftell($word);
201  while (!feof($word)) {
202  $bytesCopied += stream_copy_to_stream(
203  $word,
204  $value,
205  0xFFFFF,
206  $bytesCopied
207  );
208  }
209  rewind($value);
210  $this->setAttribute($prefix, $value);
211  continue;
212  }
213  if ('.' === $ind && 'tag' === $prefix) {
214  $this->setTag(stream_get_contents($word, -1, -1));
215  continue;
216  }
217  rewind($word);
218  $this->unrecognizedWords[] = $word;
219  }
220  } else {
221  for ($word = $com->getNextWord(); '' !== $word;
222  $word = $com->getNextWord()) {
223  if (preg_match('/^=([^=]+)=(.*)$/sS', $word, $matches)) {
224  $this->setAttribute($matches[1], $matches[2]);
225  } elseif (preg_match('/^\.tag=(.*)$/sS', $word, $matches)) {
226  $this->setTag($matches[1]);
227  } else {
228  $this->unrecognizedWords[] = $word;
229  }
230  }
231  }
232  }
