65  * Parses a value from a RouterOS scripting context.
66  *
67  * Turns a value from RouterOS into an equivalent PHP value, based on
68  * determining the type in the same way RouterOS would determine it for a
69  * literal.
70  *
71  * This method is intended to be the very opposite of
72  * {@link static::escapeValue()}. That is, results from that method, if
73  * given to this method, should produce equivalent results.
74  *
75  * @param string $value The value to be parsed.
76  * Must be a literal of a value,
77  * e.g. what {@link static::escapeValue()} will give you.
78  * @param DateTimeZone|null $timezone The timezone which any resulting
79  * DateTime object (either the main value, or values within an array)
80  * will use. Defaults to UTC.
81  *
82  * @return mixed Depending on RouterOS type detected:
83  * - "nil" (the string "[]") or "nothing" (empty string) - NULL.
84  * - "num" - int or double for large values.
85  * - "bool" - a boolean.
86  * - "array" - an array, with the keys and values processed recursively.
87  * - "time" - a {@link DateInterval} object.
88  * - "date" (pseudo type; string in the form "M/j/Y") - a DateTime
89  * object with the specified date, at midnight.
90  * - "datetime" (pseudo type; string in the form "M/j/Y H:i:s") - a
91  * DateTime object with the specified date and time.
92  * - "str" (a quoted string) - a string, with the contents escaped.
93  * - Unrecognized type - casted to a string, unmodified.
94  */
95  public static function parseValue($value, DateTimeZone $timezone = null)
96  {
97  $value = static::parseValueToSimple($value);
98  if (!is_string($value)) {
99  return $value;
100  }
101 
102  try {
103  return static::parseValueToArray($value, $timezone);
104  } catch (ParserException $e) {
105  try {
106  return static::parseValueToDateInterval($value);
107  } catch (ParserException $e) {
108  try {
109  return static::parseValueToDateTime($value, $timezone);
110  } catch (ParserException $e) {

112  }
113  }

331  * Parses a RouterOS value into a PHP array.
332  *
333  * Parses a RouterOS value into a PHP array.
334  *
335  * @param string $value The value to be parsed.
336  * Must be a literal of a value,
337  * e.g. what {@link static::escapeValue()} will give you.
338  * @param DateTimeZone|null $timezone The timezone which any resulting
339  * DateTime object within the array will use. Defaults to UTC.
340  *
341  * @return array An array, with the keys and values processed recursively,
342  * the keys with {@link static::parseValueToSimple()},
343  * and the values with {@link static::parseValue()}.
344  *
345  * @throws ParserException When the value is not of a recognized type.
346  */
347  public static function parseValueToArray(
348  $value,
349  DateTimeZone $timezone = null
350  ) {
351  $value = (string)$value;
352  if ('{' === $value[0] && '}' === $value[strlen($value) - 1]) {
353  $value = substr($value, 1, -1);
354  if ('' === $value) {
355  return array();
356  }
357  $parsedValue = preg_split(
358  '/
359  (\"(?:\\\\\\\\|\\\\"|[^"])*\")
360  |
361  (\{[^{}]*(?2)?\})
362  |
363  ([^;=]+)
364  /sx',
365  $value,
366  null,
367  PREG_SPLIT_DELIM_CAPTURE
368  );
369  $result = array();
370  $newVal = null;
371  $newKey = null;
372  for ($i = 0, $l = count($parsedValue); $i < $l; ++$i) {
373  switch ($parsedValue[$i]) {
374  case '':
375  break;
376  case ';':
377  if (null === $newKey) {
378  $result[] = $newVal;
379  } else {
380  $result[$newKey] = $newVal;
381  }
382  $newKey = $newVal = null;
383  break;
384  case '=':
385  $newKey = static::parseValueToSimple($parsedValue[$i - 1]);
386  $newVal = static::parseValue($parsedValue[++$i], $timezone);
387  break;
388  default:
389  $newVal = static::parseValue($parsedValue[$i], $timezone);
390  }
391  }
392  if (null === $newKey) {
393  $result[] = $newVal;
394  } else {
395  $result[$newKey] = $newVal;
396  }
397  return $result;
398  }
399  throw new ParserException(
400  'The supplied value can not be converted to an array',
401  ParserException::CODE_ARRAY
402  );

406  * Prepares a script.
407  *
408  * Prepares a script for eventual execution by prepending parameters as
409  * variables to it.
410  *
411  * This is particularly useful when you're creating scripts that you don't
412  * want to execute right now (as with {@link Util::exec()}, but instead
413  * you want to store it for later execution, perhaps by supplying it to
414  * "/system scheduler".
415  *
416  * @param string|resource $source The source of the script,
417  * as a string or stream. If a stream is provided, reading starts from
418  * the current position to the end of the stream, and the pointer stays
419  * at the end after reading is done.
420  * @param array<string|int,mixed> $params An array of parameters to make
421  * available in the script as local variables.
422  * Variable names are array keys, and variable values are array values.
423  * Array values are automatically processed with
424  * {@link static::escapeValue()}. Streams are also supported, and are
425  * processed in chunks, each with
426  * {@link static::escapeString()} with all bytes being escaped.
427  * Processing starts from the current position to the end of the stream,
428  * and the stream's pointer is left untouched after the reading is done.
429  * Variables with a value of type "nothing" can be declared with a
430  * numeric array key and the variable name as the array value
431  * (that is casted to a string).
432  *
433  * @return resource A new PHP temporary stream with the script as contents,
434  * with the pointer back at the start.
435  *
436  * @see static::append()
437  */
438  public static function prepare(
439  $source,
440  array $params = array()
441  ) {
442  $resultStream = fopen('php://temp', 'r+b');
443  static::append($resultStream, $source, $params);
444  rewind($resultStream);
445  return $resultStream;

448  /**
449  * Appends a script.
450  *
451  * Appends a script to an existing stream.
452  *
453  * @param resource $stream An existing stream to write the
454  * resulting script to.
455  * @param string|resource $source The source of the script,
456  * as a string or stream. If a stream is provided, reading starts from
457  * the current position to the end of the stream, and the pointer stays
458  * at the end after reading is done.
459  * @param array<string|int,mixed> $params An array of parameters to make
460  * available in the script as local variables.
461  * Variable names are array keys, and variable values are array values.
462  * Array values are automatically processed with
463  * {@link static::escapeValue()}. Streams are also supported, and are
464  * processed in chunks, each with
465  * {@link static::escapeString()} with all bytes being escaped.
466  * Processing starts from the current position to the end of the stream,
467  * and the stream's pointer is left untouched after the reading is done.
468  * Variables with a value of type "nothing" can be declared with a
469  * numeric array key and the variable name as the array value
470  * (that is casted to a string).
471  *
472  * @return int The number of bytes written to $stream is returned,
473  * and the pointer remains where it was after the write
474  * (i.e. it is not seeked back, even if seeking is supported).
475  */
476  public static function append(
477  $stream,
478  $source,
479  array $params = array()
480  ) {
481  $writer = new Stream($stream, false);
482  $bytes = 0;
483 
484  foreach ($params as $pname => $pvalue) {
485  if (is_int($pname)) {
486  $pvalue = static::escapeString((string)$pvalue);
487  $bytes += $writer->send(":local \"{$pvalue}\";\n");
488  continue;
489  }
490  $pname = static::escapeString($pname);
491  $bytes += $writer->send(":local \"{$pname}\" ");
492  if (Stream::isStream($pvalue)) {
493  $reader = new Stream($pvalue, false);
494  $chunkSize = $reader->getChunk(Stream::DIRECTION_RECEIVE);
495  $bytes += $writer->send('"');
496  while ($reader->isAvailable() && $reader->isDataAwaiting()) {
497  $bytes += $writer->send(
498  static::escapeString(fread($pvalue, $chunkSize), true)
499  );
500  }
501  $bytes += $writer->send("\";\n");
502  } else {
503  $bytes += $writer->send(static::escapeValue($pvalue) . ";\n");
504  }
505  }
506 
507  $bytes += $writer->send($source);
508  return $bytes;
509  }

511  /**
512  * Escapes a value for a RouterOS scripting context.
513  *
514  * Turns any native PHP value into an equivalent whole value that can be
515  * inserted as part of a RouterOS script.
516  *
517  * DateInterval objects will be casted to RouterOS' "time" type.
518  *
519  * DateTime objects will be casted to a string following the "M/d/Y H:i:s"
520  * format. If the time is exactly midnight (including microseconds), and
521  * the timezone is UTC, the string will include only the "M/d/Y" date.
522  *
523  * Unrecognized types (i.e. resources and other objects) are casted to
524  * strings, and those strings are then escaped.
525  *
526  * @param mixed $value The value to be escaped.
527  *
528  * @return string A string representation that can be directly inserted in a
529  * script as a whole value.
530  */
531  public static function escapeValue($value)
532  {
533  switch(gettype($value)) {
534  case 'NULL':
535  $value = '[]';
536  break;
537  case 'integer':
538  $value = (string)$value;
539  break;
540  case 'boolean':
541  $value = $value ? 'true' : 'false';
542  break;
543  case 'array':
544  if (0 === count($value)) {
545  $value = '({})';
546  break;
547  }
548  $result = '';
549  foreach ($value as $key => $val) {
550  $result .= ';';
551  if (!is_int($key)) {
552  $result .= static::escapeValue($key) . '=';
553  }
554  $result .= static::escapeValue($val);
555  }
556  $value = '{' . substr($result, 1) . '}';
557  break;
558  case 'object':
559  if ($value instanceof DateTime) {
560  $usec = $value->format('u');
561  $usec = '000000' === $usec ? '' : '.' . $usec;
562  $value = '00:00:00.000000 UTC' === $value->format('H:i:s.u e')
563  ? $value->format('M/d/Y')
564  : $value->format('M/d/Y H:i:s') . $usec;
565  }
566  if ($value instanceof DateInterval) {
567  if (false === $value->days || $value->days < 0) {
568  $value = $value->format('%r%dd%H:%I:%S');
569  } else {
570  $value = $value->format('%r%ad%H:%I:%S');
571  }
572  break;
573  }
574  //break; intentionally omitted
575  default:
576  $value = '"' . static::escapeString((string)$value) . '"';
577  break;
578  }
579  return $value;
580  }

582  /**
583  * Escapes a string for a RouterOS scripting context.
584  *
585  * Escapes a string for a RouterOS scripting context. The value can then be
586  * surrounded with quotes at a RouterOS script (or concatenated onto a
587  * larger string first), and you can be sure there won't be any code
588  * injections coming from it.
589  *
590  * By default, for the sake of brevity of the output, ASCII alphanumeric
591  * characters and underscores are left untouched. And for the sake of
592  * character conversion, bytes above 0x7F are also left untouched.
593  *
594  * @param string $value Value to be escaped.
595  * @param bool $full Whether to escape all bytes in the string, including
596  * ASCII alphanumeric characters, underscores and bytes above 0x7F.
597  *
598  * @return string The escaped value.
599  *
600  * @internal Why leave ONLY those ASCII characters and not also others?
601  * Because those can't in any way be mistaken for language constructs,
602  * unlike many other "safe inside strings, but not outside" ASCII
603  * characters, like ",", ".", "+", "-", "~", etc.
604  */
605  public static function escapeString($value, $full = false)
606  {
607  if ($full) {
608  return self::_escapeCharacters(array($value));
609  }
610  return preg_replace_callback(
611  '/[^\\_A-Za-z0-9\\x80-\\xFF]+/S',
612  array(__CLASS__, '_escapeCharacters'),
613  $value
614  );
615  }
