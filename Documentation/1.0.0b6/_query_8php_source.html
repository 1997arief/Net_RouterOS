203  $old = $com->getTransmitter()->lock(T\Stream::DIRECTION_SEND);
204  $bytes = $this->_send($com);

211  /**
212  * Sends the query over a communicator.
213  *
214  * The only difference with the non private equivalent is that this one does
215  * not do locking.
216  *
217  * @param Communicator $com The communicator to send the query over.
218  *
219  * @return int The number of bytes sent.
220  */
221  private function _send(Communicator $com)
222  {
223  if (!$com->getTransmitter()->isAcceptingData()) {
224  throw new SocketException(
225  'Transmitter is invalid. Sending aborted.',
226  SocketException::CODE_QUERY_SEND_FAIL
227  );
228  }
229  $bytes = 0;
230  foreach ($this->words as $queryWord) {
231  list($predicate, $value) = $queryWord;
232  $prefix = '?' . $predicate;
233  if (null === $value) {
234  $bytes += $com->sendWord($prefix);
235  } else {
236  $prefix .= '=';
237  if (is_string($value)) {
238  $bytes += $com->sendWord($prefix . $value);
239  } else {
240  $bytes += $com->sendWordFromStream($prefix, $value);
241  }
242  }
243  }
244  return $bytes;
245  }

247  /**
248  * Verifies the query.
249  *
250  * Verifies the query against a communicator, i.e. whether the query
251  * could successfully be sent (assuming the connection is still opened).
252  *
253  * @param Communicator $com The Communicator to check against.
254  *
255  * @return $this The query object itself.
256  *
257  * @throws LengthException If the resulting length of an API word is not
258  * supported.
259  */
260  public function verify(Communicator $com)
261  {
262  foreach ($this->words as $queryWord) {
263  list($predicate, $value) = $queryWord;
264  if (null === $value) {
265  $com::verifyLengthSupport(strlen('?' . $predicate));
266  } elseif (is_string($value)) {
267  $com::verifyLengthSupport(
268  strlen('?' . $predicate . '=' . $value)
269  );
270  } else {
271  $com::verifyLengthSupport(
272  strlen('?' . $predicate . '=') +
273  $com::seekableStreamLength($value)
274  );
275  }
276  }
277  return $this;
278  }
