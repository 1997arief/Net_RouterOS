115  * Sets the command to send to RouterOS.
116  *
117  * Sets the command to send to RouterOS. The command can use the API or CLI
118  * syntax of RouterOS, but either way, it must be absolute (begin with a
119  * "/") and without arguments.
120  *
121  * @param string $command The command to send.
122  *
123  * @return $this The request object.
124  *
125  * @see getCommand()
126  * @see setArgument()
127  */
128  public function setCommand($command)
129  {
130  $command = (string) $command;
131  if (strpos($command, '/') !== 0) {
132  throw new InvalidArgumentException(
133  'Commands must be absolute.',
134  InvalidArgumentException::CODE_ABSOLUTE_REQUIRED
135  );
136  }
137  if (substr_count($command, '/') === 1) {
138  //Command line syntax convertion
139  $cmdParts = preg_split('#[\s/]+#sm', $command);
140  $cmdRes = array($cmdParts[0]);
141  for ($i = 1, $n = count($cmdParts); $i < $n; $i++) {
142  if ('..' === $cmdParts[$i]) {
143  $delIndex = count($cmdRes) - 1;
144  if ($delIndex < 1) {
145  throw new InvalidArgumentException(
146  'Unable to resolve command',
147  InvalidArgumentException::CODE_CMD_UNRESOLVABLE
148  );
149  }
150  unset($cmdRes[$delIndex]);
151  $cmdRes = array_values($cmdRes);
152  } else {
153  $cmdRes[] = $cmdParts[$i];
154  }
155  }
156  $command = implode('/', $cmdRes);
157  }
158  if (!preg_match('#^/\S+$#sm', $command)) {
159  throw new InvalidArgumentException(
160  'Invalid command supplied.',
161  InvalidArgumentException::CODE_CMD_INVALID
162  );
163  }
164  $this->_command = $command;
165  return $this;

275  * Sends a request over a communicator.
276  *
277  * @param Communicator $com The communicator to send the request over.
278  * @param Registry|null $reg An optional registry to sync the request with.

285  public function send(Communicator $com, Registry $reg = null)
286  {
287  if (null !== $reg
288  && (null != $this->getTag() || !$reg->isTaglessModeOwner())
289  ) {
290  $originalTag = $this->getTag();
291  $this->setTag($reg->getOwnershipTag() . $originalTag);
292  $bytes = $this->send($com);
293  $this->setTag($originalTag);
294  return $bytes;
295  }
296  if ($com->getTransmitter()->isPersistent()) {
297  $old = $com->getTransmitter()->lock(T\Stream::DIRECTION_SEND);
298  $bytes = $this->_send($com);
299  $com->getTransmitter()->lock($old, true);
300  return $bytes;
301  }
302  return $this->_send($com);
303  }

305  /**
306  * Sends a request over a communicator.
307  *
308  * The only difference with the non private equivalent is that this one does
309  * not do locking.
310  *
311  * @param Communicator $com The communicator to send the request over.
312  *
313  * @return int The number of bytes sent.
314  *
315  * @see Client::sendSync()
316  * @see Client::sendAsync()
317  */
318  private function _send(Communicator $com)
319  {
320  if (!$com->getTransmitter()->isAcceptingData()) {
321  throw new SocketException(
322  'Transmitter is invalid. Sending aborted.',
323  SocketException::CODE_REQUEST_SEND_FAIL
324  );
325  }
326  $bytes = 0;
327  $bytes += $com->sendWord($this->getCommand());
328  if (null !== ($tag = $this->getTag())) {
329  $bytes += $com->sendWord('.tag=' . $tag);
330  }
331  foreach ($this->attributes as $name => $value) {
332  $prefix = '=' . $name . '=';
333  if (is_string($value)) {
334  $bytes += $com->sendWord($prefix . $value);
335  } else {
336  $bytes += $com->sendWordFromStream($prefix, $value);
337  }
338  }
339  $query = $this->getQuery();
340  if ($query instanceof Query) {
341  $bytes += $query->send($com);
342  }
343  $bytes += $com->sendWord('');
344  return $bytes;
345  }

381  /**
382  * Parses the arguments of a command.
383  *
384  * @param string $string The argument string to parse.
385  *
386  * @return void
387  */
388  protected function parseArgumentString($string)
389  {
390  /*
391  * Grammar:
392  *
393  * <arguments> := (<<\s+>>, <argument>)*,
394  * <argument> := <name>, <value>?
395  * <name> := <<[^\=\s]+>>
396  * <value> := "=", (<quoted string> | <unquoted string>)
397  * <quotedString> := <<">>, <<([^"]|\\"|\\\\)*>>, <<">>
398  * <unquotedString> := <<\S+>>
399  */
400 
401  $token = '';
402  $name = null;
403  while ($string = substr($string, strlen($token))) {
404  if (null === $name) {
405  if (preg_match('/^\s+([^\s=]+)/sS', $string, $matches)) {
406  $token = $matches[0];
407  $name = $matches[1];
408  } else {
409  throw new InvalidArgumentException(
410  "Parsing of argument name failed near '{$string}'",
411  InvalidArgumentException::CODE_NAME_UNPARSABLE
412  );
413  }
414  } elseif (preg_match('/^\s/s', $string, $matches)) {
415  //Empty argument
416  $token = '';
417  $this->setArgument($name);
418  $name = null;
419  } elseif (preg_match(
420  '/^="(([^\\\"]|\\\"|\\\\)*)"/sS',
421  $string,
422  $matches
423  )) {
424  $token = $matches[0];
425  $this->setArgument(
426  $name,
427  str_replace(
428  array('\\"', '\\\\'),
429  array('"', '\\'),
430  $matches[1]
431  )
432  );
433  $name = null;
434  } elseif (preg_match('/^=(\S+)/sS', $string, $matches)) {
435  $token = $matches[0];
436  $this->setArgument($name, $matches[1]);
437  $name = null;
438  } else {
439  throw new InvalidArgumentException(
440  "Parsing of argument value failed near '{$string}'",
441  InvalidArgumentException::CODE_VALUE_UNPARSABLE
442  );
443  }
444  }
445 
446  if (null !== $name && ('' !== ($name = trim($name)))) {
447  $this->setArgument($name, '');
448  }
449 
450  }
